00072 /* Please report any problems that you find with your compiler, which may
00073  * be solved in this section! There are several compilers that I have not
00074  * been able to test this file with yet.
00075  *
00076  * First: If we are we on Windows, we want a single define for it (_WIN32)
00077  * (Note: For Cygwin the compiler flag -mwin32 should be used, but to
00078  * make sure that things run smoothly for Cygwin users, we add __CYGWIN__
00079  * to the list of "valid Win32 identifiers", which removes the need for
00080  * -mwin32)
00081  */
00082 #if !defined(_WIN32) && (defined(__WIN32__) || defined(WIN32) || defined(__CYGWIN__))
00083  #define _WIN32
00084 #endif /* _WIN32 */

00086 /* In order for extension support to be portable, we need to define an
00087  * OpenGL function call method. We use the keyword APIENTRY, which is
00088  * defined for Win32. (Note: Windows also needs this for <GL/gl.h>)
00089  */
00090 #ifndef APIENTRY
00091  #ifdef _WIN32
00092   #define APIENTRY __stdcall
00093  #else
00094   #define APIENTRY
00095  #endif
00096 #endif /* APIENTRY */

00099  * <GL/gl.h> files happy. Theoretically we could include <windows.h>, but
00100  * it has the major drawback of severely polluting our namespace.
00101  */
00102 
00103 /* Under Windows, we need WINGDIAPI defined */
00104 #if !defined(WINGDIAPI) && defined(_WIN32)
00105  #if defined(_MSC_VER) || defined(__BORLANDC__) || defined(__POCC__)
00106   /* Microsoft Visual C++, Borland C++ Builder and Pelles C */
00107   #define WINGDIAPI __declspec(dllimport)
00108  #elif defined(__LCC__)
00109   /* LCC-Win32 */
00110   #define WINGDIAPI __stdcall
00111  #else
00112   /* Others (e.g. MinGW, Cygwin) */
00113   #define WINGDIAPI extern
00114  #endif
00115  #define GLFW_WINGDIAPI_DEFINED
00116 #endif /* WINGDIAPI */
00117 
00118 /* Some <GL/glu.h> files also need CALLBACK defined */
00119 #if !defined(CALLBACK) && defined(_WIN32)
00120  #if defined(_MSC_VER)
00121   /* Microsoft Visual C++ */
00122   #if (defined(_M_MRX000) || defined(_M_IX86) || defined(_M_ALPHA) || defined(_M_PPC)) && !defined(MIDL_PASS)
00123    #define CALLBACK __stdcall
00124   #else
00125    #define CALLBACK
00126   #endif
00127  #else
00128   /* Other Windows compilers */
00129   #define CALLBACK __stdcall
00130  #endif
00131  #define GLFW_CALLBACK_DEFINED
00132 #endif /* CALLBACK */
